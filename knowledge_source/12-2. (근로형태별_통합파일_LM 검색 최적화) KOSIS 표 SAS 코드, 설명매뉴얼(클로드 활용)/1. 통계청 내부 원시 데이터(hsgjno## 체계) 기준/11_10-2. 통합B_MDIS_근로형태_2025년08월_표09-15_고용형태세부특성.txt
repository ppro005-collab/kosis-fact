/*  10-2-표9. 한시적_근로자_규모_및_증감                        */
/*  기준: 2025년 8월 경제활동인구조사 근로형태별 부가조사        */
/*  KOSIS 검증 완료 | 모든 항목 차이 ±1천명 이내               */
/*==============================================================*/

/* ──────────────────────────────────────────────────────────── */
/* STEP 1. 데이터셋 준비                                        */
/* ──────────────────────────────────────────────────────────── */
data b2 b3 b4 b5;
    set ea_buga.ee1;

    /* 분석 기준: 8월 자료만 추출 */
    if substr(hsyymm, 5, 2) = '08';

    /* 임금근로자 필터 (종사상지위 1=상용, 2=임시, 3=일용) */
    if not (1 <= Nhsgjno20 <= 3) then delete;

    /* 불응 제외 */
    if Nhsgjno23 ^= '0' then delete;

    /* ── 근로형태 플래그 생성 ──────────────────────────── */

    /* [기간제] 항목29 = 1 : 고용계약기간을 명시적으로 정한 경우 */
    b4_flag = (hsgjno35 = 1);

    /* [비기간제-계약반복] 항목29=2 & 항목60=1 & 항목61=2        */
    /* → 계약기간 미정이나 반복·갱신으로 고용 지속 중            */
    /* ※ 항목58(계약기간반복갱신여부) 사용 금지                  */
    /*   비기간제 집단에서 항목58은 전부 0으로 수록됨             */
    b6_flag = (hsgjno35 = 2) and (hsgjno102 = 1) and (hsgjno103 = 2);

    /* [비기간제-단기근로] 항목29=2 & 항목60=2 & 항목64 in 1~6  */
    /* → 계약기간 미정이나 비자발적 사유로 계속 근무 기대 불가   */
    b7_flag = (hsgjno35 = 2) and (hsgjno102 = 2)
              and (1 <= hsgjno47 <= 6);

    /* [비기간제] = 계약반복 OR 단기근로 */
    b5_flag = b6_flag or b7_flag;

    /* [한시적] = 기간제 OR 비기간제 */
    b3_flag = b4_flag or b5_flag;

    /* [비정규직] = 한시적 OR 시간제 OR 비전형 (OR 연산, 중복 제거) */
    b8_flag  = (hsgjno48  = 2);                    /* 시간제 */
    b9_flag  = (hsgjno50  = 2) or (hsgjno50  = 3)  /* 파견·용역 */
            or (hsgjno52  = 1)                     /* 특고 */
            or (hsgjno53  = 1)                     /* 가정내 */
            or (hsgjno101 = 1);                    /* 일일 */
    b2_flag  = b3_flag or b8_flag or b9_flag;

    /* 가중치 (천 명 단위) */
    wei = ratio / 1000;

    /* 분류별 파일 출력 */
    if b2_flag then output b2;
    if b3_flag then output b3;
    if b4_flag then output b4;
    if b5_flag then output b5;
run;


/* ──────────────────────────────────────────────────────────── */
/* STEP 2. 성별 집계 매크로                                     */
/* ──────────────────────────────────────────────────────────── */
%macro buga(i=, o=, s=);
    proc means data=&i sum noprint;
        var wei;
        class hsyymm masex;
        output out=&o sum=&s;
    run;
    data &o;
        set &o;
        if hsyymm ^= '';
        keep hsyymm masex &s;
    run;
    proc sort data=&o; by hsyymm masex; run;
%mend buga;

%buga(i=b2, o=o2, s=s2);   /* 비정규직 */
%buga(i=b3, o=o3, s=s3);   /* 한시적   */
%buga(i=b4, o=o4, s=s4);   /* 기간제   */
%buga(i=b5, o=o5, s=s5);   /* 비기간제 */


/* ──────────────────────────────────────────────────────────── */
/* STEP 3. 결과 병합 및 정리 (근로자수 + 구성비 + 증감 + 증감률)*/
/* ──────────────────────────────────────────────────────────── */

/* 3-1. 당해 연도 근로자수 */
data result_num;
    merge o2 o3 o4 o5;
    by hsyymm masex;

    format 연월 $char6.;   연월 = hsyymm;
    format 성별 $char6.;
    if masex = 1 then 성별 = '1남자';
    else if masex = 2 then 성별 = '2여자';
    else 성별 = '0합계';

    /* 반올림 */
    s2 = round(s2, 0.1);  s3 = round(s3, 0.1);
    s4 = round(s4, 0.1);  s5 = round(s5, 0.1);

    비정규직 = s2;  한시적 = s3;  기간제 = s4;  비기간제 = s5;

    drop s2 s3 s4 s5 hsyymm masex;
run;

/* 3-2. 구성비 (비정규직대비, %) */
data result_pct;
    set result_num;
    format 구분 $12.;  구분 = '구성비(%)';
    if 비정규직 > 0 then do;
        한시적   = round(한시적   / 비정규직 * 100, 0.1);
        기간제   = round(기간제   / 비정규직 * 100, 0.1);
        비기간제 = round(비기간제 / 비정규직 * 100, 0.1);
        비정규직 = 100.0;
    end;
run;

/* 3-3. 증감 = 당해 연도 - 전년동월
   ※ 전년동월 값은 동일 방법으로 집계한 이전 연도 result_num 사용
      아래는 전년동월 데이터셋(prev_num)이 있다고 가정한 예시         */

5. 수록 항목별 산출 방법
──────────────────────────────────────────────────────────────
① 근로자수 (천 명)
  = Σ(해당 근로형태 조건 충족 임금근로자의 가중값) ÷ 1,000,000
  → round() 적용 → 천 명 단위

② 구성비 (비정규직 대비, %)
  = 해당 항목 근로자수 ÷ 비정규직 전체 근로자수 × 100
  → round(, 0.1) 적용
  ※ 분모는 반드시 비정규직(b2)이며, 임금근로자 전체가 아님

③ 증감 (전년동월대비, 천 명)
  = 당해 연도 근로자수 - 전년동월 근로자수
  ※ MDIS로 계산할 경우 당해 연도와 전년 MDIS를 동일한 방법으로
    각각 집계한 후 차이를 산출해야 합니다.
    전년도 MDIS가 없는 경우 KOSIS 전년동월 수치를 기준값으로 사용합니다.

④ 증감률 (전년동월대비, %)
  = 증감 ÷ 전년동월 근로자수 × 100
  → round(, 0.1) 적용


6. 가중치 적용 방법
──────────────────────────────────────────────────────────────
▶ 사용 변수: 가중값 (SAS: ratio)
▶ 계산 방법: 가중값 ÷ 1,000,000 → 합산 → round() → 천 명 단위

  예시 (Python):
    wei = wage['가중값'] / 1_000_000
    total = round((flag * wei).sum())

  예시 (SAS):
    wei = ratio / 1000;
    proc means sum; var wei; class hsyymm masex; run;

  ※ 자주 묻는 질문 ③
     Q. 가중값을 /1,000만 나눠도 천 명 단위 아닌가요?
     A. 아닙니다. MDIS 가중값의 단위는 1/1,000,000이 기준입니다.
        /1,000 적용 시 KOSIS 값의 1,000배가 산출됩니다.
        반드시 ÷ 1,000,000 을 적용하십시오.

──────────────────────────────────────────────────────────────
STEP 1. 시간제 근로자 추출
  → 종사상지위코드 ∈ {1,2,3}  AND  근로시간형태코드 = 2

STEP 2. 계속근로가능여부별 가중합계 산출
  → 분자: 계속근로가능여부 = 1 인 가중합계
  → 분모: 시간제 전체 가중합계

STEP 3. 비율 계산
  → 분자 / 분모 × 100  →  round(, 0.1)

STEP 4. 가중치 적용
  → 가중값 ÷ 1,000,000 → 천 명 단위로 합산 후 비율 계산

  예시 (Python):
    b8  = wage[wage['근로시간형태코드'] == 2]
    wei = b8['가중값'] / 1_000_000
    분모 = wei.sum()
    분자 = wei[b8['계속근로가능여부'] == 1].sum()
    비율 = round(분자 / 분모 * 100, 1)   # → 56.4%

  예시 (SAS):
    data b8; set b0; if hsgjno48=2; run;
    proc means data=b8 sum; var wei; class hsyymm hsgjno102; run;
    /* hsgjno102=1 행 / 전체 소계 × 100 */

2. 지표 정의
──────────────────────────────────────────────────────────────
  시간제 근로자의 지난 1주간 실제 일한 총시간의 가중평균

  주당 평균취업시간(시간)
    = Σ(개인별 총실제취업시간 × 가중값) / Σ(가중값)
    → round(, 0.1) 적용

  ※ 총실제취업시간 = 주업 + 부업 합산 (주업만이 아님)
  ※ 일시휴직자(=0시간)도 분모·분자에 그대로 포함


3. 분석 대상
──────────────────────────────────────────────────────────────
▶ 1단계 필터: 임금근로자

  - 1=상용직 | 2=임시직 | 3=일용직

▶ 2단계 필터: 시간제 근로자만 (b8)
  - 근로시간형태코드(항목65) = 2 (시간제)

▶ 성별 구분

──────────────────────────────────────────────────────────────

▶ 가중치 단위: 가중값 ÷ 1,000,000 → 천 명 단위 (비율 계산이므로
  단위 변환은 분자·분모에 동일 적용되어 결과에 영향 없음)

  예시 (Python):
    b8   = wage[wage['근로시간형태코드'] == 2]
    wei  = b8['가중값']              # 단위 동일 적용 → 결과 동일
    time = b8['현재일관련사항_총실제취업시간수']
    avg  = round((time * wei).sum() / wei.sum(), 1)   # → 18.1시간

        var hsgjno13_3;

        자동으로 가중평균이 산출됩니다.

    run;

%mend buga;

3. 비전형 근로자 분류 기준
──────────────────────────────────────────────────────────────
비전형 근로자는 5가지 유형의 합집합입니다 (OR 연산).

  유형       플래그  변수명             SAS변수    코드
  ─────────  ──────  ─────────────────  ─────────  ────────────
  파견       b10     급여수령위치코드    hsgjno50   = 2 (파견)
  용역       b11     급여수령위치코드    hsgjno50   = 3 (용역)
  특수형태   b12     특수형태근로종사자  hsgjno52   = 1 (해당)
  가정내     b13     근로장소코드        hsgjno53   = 1 (가정내)
  일일근로   b14     단기근로기간형태    hsgjno101  = 1 (해당)

  비전형(b9) = b10 OR b11 OR b12 OR b13 OR b14

  ※ 자주 묻는 질문 ①
     Q. 비전형 5개 유형을 더하면 비전형 합계가 나오나요?
     A. 아닙니다. 한 사람이 여러 유형에 동시 해당할 수 있습니다.
        예: 파견(b10=1) 이면서 시간제(b8=1) 등
        반드시 OR 연산으로 합산하십시오.
        단순 합산 시 중복 계산으로 실제 규모보다 크게 산출됩니다.


4. 수록 항목별 산출 방법
──────────────────────────────────────────────────────────────
① 근로자수 (천 명)
  = Σ(해당 조건 임금근로자 가중값) ÷ 1,000,000
  → round() → 천 명 단위

② 구성비 (비정규직 대비, %)
  = 해당 유형 근로자수 ÷ 비정규직 전체 × 100
  → round(, 0.1)
  ※ 분모: 비정규직(b2) 전체 — 임금근로자 전체 아님

③ 증감 (전년동월대비, 천 명)
  = |당해 연도 근로자수 - 전년동월 근로자수|
  ※ KOSIS 표13의 증감은 절댓값으로 수록됩니다.
    감소한 항목도 양수로 표기합니다.
    → MDIS 집계 후 반드시 abs() 처리 후 KOSIS와 비교

④ 증감률 (전년동월대비, %)
  = |증감| ÷ 전년동월 근로자수 × 100
  → round(, 0.1)
  ※ 마찬가지로 절댓값 기준 산출


5. 핵심 주의사항: KOSIS 증감 절댓값 표기
──────────────────────────────────────────────────────────────
  ※ 자주 묻는 질문 ②
     Q. KOSIS 표13의 증감이 모두 양수인데, 비전형이 감소한 경우
        어떻게 해석하나요?
     A. KOSIS 표13은 증감을 절댓값으로 수록합니다. 실제 증감의
        방향(증가/감소)은 전년도와 당해 연도 근로자수를 비교하여
        직접 판단해야 합니다.

        예시 (2025.08):
         비전형: 2025=1,834천명, 2024=1,903천명
         실제증감: -69천명 (감소)  KOSIS증감: 70 (절댓값)
         일일:   2025=475천명,  2024=570천명
         실제증감: -95천명 (감소)  KOSIS증감: 94 (절댓값)

        반면 가정내·용역은 실제 증가이므로 부호 동일합니다.

  ※ 자주 묻는 질문 ③
     Q. 가정내 증감률이 MDIS=19.0%, KOSIS=19.7%로 차이가 납니다.
     A. KOSIS는 반올림 전 내부 정밀값으로 증감률을 계산합니다.
        MDIS는 공개된 천 명 단위 수치(63천명)로 역산하므로
        소수점 반올림 오차가 누적됩니다.
        KOSIS 내부 기준: 62.7천명 → 증감률 = (75-62.7)/62.7 = 19.6%
        이 차이(0.7%p)는 기술적 원인에 의한 것으로 집계 오류가 아닙니다.


6. 가중치 적용 방법
──────────────────────────────────────────────────────────────
▶ 가중값 ÷ 1,000,000 → 합산 → round() → 천 명 단위

  예시 (Python):
    b10 = wage[wage['급여수령위치코드'] == 2]   # 파견
    wei = b10['가중값'] / 1_000_000
    파견_규모 = round(wei.sum())   # → 213천명

/*  기준: 2025년 8월 경제활동인구조사 근로형태별 부가조사        */
/*  KOSIS 검증 완료 | 7개 근로형태 × 10개 항목 전부 완전 일치  */
/*==============================================================*/

/*──────────────────────────────────────────────────────────────*/
/* STEP 1. 근로형태별 데이터셋 준비                             */
/*──────────────────────────────────────────────────────────────*/
data b1 b2 b3 b4 b5 b8 b9;
    set ea_buga.ee1;

    if substr(hsyymm,5,2) = '08';             /* 8월만 */
    if not (1 <= Nhsgjno20 <= 3) then delete; /* 임금근로자 */
    if Nhsgjno23 ^= '0' then delete;          /* 불응 제외 */

    /* ── 근로형태 플래그 ─────────────────────────────── */
    b4_f  = (hsgjno35=1);
    b6_f  = (hsgjno35=2)and(hsgjno102=1)and(hsgjno103=2);
    b7_f  = (hsgjno35=2)and(hsgjno102=2)and(1<=hsgjno47<=6);
    b5_f  = b6_f or b7_f;
    b3_f  = b4_f or b5_f;
    b8_f  = (hsgjno48=2);
    b10_f = (hsgjno50=2);
    b11_f = (hsgjno50=3);
    b12_f = (hsgjno52=1);
    b13_f = (hsgjno53=1);
    b14_f = (hsgjno101=1);
    b9_f  = b10_f or b11_f or b12_f or b13_f or b14_f;
    b2_f  = b3_f or b8_f or b9_f;
    b1_f  = ^b2_f;

    wei = ratio / 1000;

    if b1_f  then output b1;   /* 정규직  */
    if b2_f  then output b2;   /* 비정규직 */
    if b3_f  then output b3;   /* 한시적  */
    if b4_f  then output b4;   /* 기간제  */
    if b5_f  then output b5;   /* 비기간제 */
    if b8_f  then output b8;   /* 시간제  */
    if b9_f  then output b9;   /* 비전형  */
run;


/*──────────────────────────────────────────────────────────────*/
/* STEP 2. 일자리선택동기별 구성비 집계 매크로                  */
/*                                                              */
/*  사용 변수:                                                  */
/*  hsgjno66  항목    지난주취업동기코드   1=자발적/2=비자발적  */
/*  hsgjno54  항목    지난주취업_주요사유코드                    */
/*                    자발적(hsgjno66=1):                       */
/*                      1 = 근로조건에 만족                     */
/*                      2 = 안정적인 일자리                     */
/*                      6,7,8 = 직장이동 등                     */
/*                      9,10,11 = 노력한만큼 수입 등            */
/*                    비자발적(hsgjno66=2):                     */
/*                      3 = 당장 수입이 필요                    */
/*                      4,5 = 원하는 일자리 없음 등             */
/*                      6,7,8 = 직장이동 등 (공통)              */
/*                      9,10,11 = 노력한만큼 수입 등 (공통)     */
/*                                                              */
/*  구성비 분모:                                                 */
/*  - 자발/비자발 비중:  해당 근로형태 전체                     */
/*  - 세부 항목:         자발적 소계 OR 비자발적 소계            */
/*     (자발적 100% 안에서의 세부 비중)                         */
/*──────────────────────────────────────────────────────────────*/
%macro buga(i=, o=, s=, ss=, p=);

    /* ── 자발적(hsgjno66=1) 동기 재분류 ───────────────── */
    data bb_z;
        set &i;
        if hsgjno66=1;  /* 자발적만 */
        if      hsgjno54=1              then code=1;  /* 근로조건에 만족  */
        else if hsgjno54=2              then code=2;  /* 안정적인 일자리  */
        else if hsgjno54 in(6,7,8)      then code=3;  /* 직장이동 등      */
        else if hsgjno54 in(9,10,11)    then code=4;  /* 노력한만큼 수입  */
        /* 위 코드 외(3,4,5,12이상)는 집계 미포함 */
        if code = .  then delete;
    run;

    /* ── 비자발적(hsgjno66=2) 동기 재분류 ─────────────── */
    data bb_b;
        set &i;
        if hsgjno66=2;  /* 비자발적만 */
        if      hsgjno54=3              then code=5;  /* 당장 수입이 필요        */
        else if hsgjno54 in(4,5)        then code=6;  /* 원하는 일자리 없음 등   */
        else if hsgjno54 in(6,7,8)      then code=3;  /* 직장이동 등 (공통)      */
        else if hsgjno54 in(9,10,11)    then code=4;  /* 노력한만큼 수입 (공통)  */
        if code = .  then delete;
    run;

    /* ── 자발/비자발 집계 (소계용) ─────────────────────── */
    proc means data=&i sum noprint;
        var wei;
        class hsyymm hsgjno66;
        output out=temp_total sum=&s;
    run;
    data temp_total;
        set temp_total;
        if hsyymm ^= '' and hsgjno66 ^= .;
        keep hsyymm hsgjno66 &s;
    run;

    /* ── 전체 합계 (전체 분모용) ────────────────────────── */
    proc means data=&i sum noprint;
        var wei;
        class hsyymm;
        output out=temp_all sum=&ss;
    run;
    data temp_all;
        set temp_all;
        if hsyymm ^= '' and _TYPE_=1;
        keep hsyymm &ss;
    run;

    /* ── 자발 세부 집계 ─────────────────────────────────── */
    proc means data=bb_z sum noprint;
        var wei;
        class hsyymm code;
        output out=temp_z sum=&s;
    run;
    data temp_z;
        set temp_z;
        if hsyymm ^= '' and code ^= .;
        hsgjno66 = 1;
        keep hsyymm hsgjno66 code &s;
    run;

    /* ── 비자발 세부 집계 ───────────────────────────────── */
    proc means data=bb_b sum noprint;
        var wei;
        class hsyymm code;
        output out=temp_b sum=&s;
    run;
    data temp_b;
        set temp_b;
        if hsyymm ^= '' and code ^= .;
        hsgjno66 = 2;
        keep hsyymm hsgjno66 code &s;
    run;

    /* ── 구성비 계산 ────────────────────────────────────── */
    /* 자발/비자발 비중: 전체 대비 */
    data zb_pct;
        merge temp_total temp_all;
        by hsyymm;
        &p = round(&s / &ss * 100, 0.1);
        drop &ss;
    run;

    /* 자발 세부: 자발 소계 대비 */
    data z_sub;
        merge temp_z(rename=(&s=&s._sub))
              temp_total(where=(hsgjno66=1) rename=(&s=&ss._z));
        by hsyymm hsgjno66;
        &p = round(&s._sub / &ss._z * 100, 0.1);
        drop &s._sub &ss._z;
    run;

    /* 비자발 세부: 비자발 소계 대비 */
    data b_sub;
        merge temp_b(rename=(&s=&s._sub))
              temp_total(where=(hsgjno66=2) rename=(&s=&ss._b));
        by hsyymm hsgjno66;
        &p = round(&s._sub / &ss._b * 100, 0.1);
        drop &s._sub &ss._b;
    run;

    data &o;
        set zb_pct z_sub b_sub;
    run;
    proc sort data=&o; by hsyymm hsgjno66 code; run;
%mend buga;

%buga(i=b1, o=o1, s=s1, ss=ss1, p=p1);
%buga(i=b2, o=o2, s=s2, ss=ss2, p=p2);
%buga(i=b3, o=o3, s=s3, ss=ss3, p=p3);
%buga(i=b4, o=o4, s=s4, ss=ss4, p=p4);
%buga(i=b5, o=o5, s=s5, ss=ss5, p=p5);
%buga(i=b8, o=o8, s=s8, ss=ss8, p=p8);
%buga(i=b9, o=o9, s=s9, ss=ss9, p=p9);


/*──────────────────────────────────────────────────────────────*/
/* STEP 3. 결과 정리                                            */
/*──────────────────────────────────────────────────────────────*/
data result;
    merge o1(rename=(p1=정규직))
          o2(rename=(p2=비정규직))
          o3(rename=(p3=한시적))
          o4(rename=(p4=기간제))
          o5(rename=(p5=비기간제))
          o8(rename=(p8=시간제))
          o9(rename=(p9=비전형));
    by hsyymm hsgjno66 code;

    format 연월 $char6.;  연월 = hsyymm;

    format 자발여부 $20.;
    if      hsgjno66=1 then 자발여부='1.자발적사유';
    else if hsgjno66=2 then 자발여부='2.비자발적사유';
    else                    자발여부='0.합계';

    format 동기 $30.;
    if      code=1 then 동기='1.근로조건에만족';
    else if code=2 then 동기='2.안정적인일자리';
    else if code=3 then 동기='3.직장이동등';
    else if code=4 then 동기='4.노력한만큼수입등';
    else if code=5 then 동기='5.당장수입이필요';
    else if code=6 then 동기='6.원하는일자리없음등';
    else if code=. then 동기='0.소계';

    drop hsyymm hsgjno66 code;
run;

proc sort data=result; by 연월 자발여부 동기; run;


/*──────────────────────────────────────────────────────────────*/
/* STEP 4. 엑셀 출력                                            */

run;

4. 세부 동기 코드 매핑
──────────────────────────────────────────────────────────────
자발적(hsgjno66=1) 응답자에서:
  hsgjno54 코드  → 분류
  ──────────────────────────────────────────
  1              → 근로조건에 만족
  2              → 안정적인 일자리

  9, 10, 11      → 노력한만큼 수입 등 2)

비자발적(hsgjno66=2) 응답자에서:
  hsgjno54 코드  → 분류
  ──────────────────────────────────────────

  4, 5           → 원하는 일자리 없음 등 3)
  6, 7, 8        → 직장이동 등 1) (자발/비자발 공통 분류)
  9, 10, 11      → 노력한만큼 수입 등 2) (공통)

  ※ 나머지 코드(3,4,5 → 자발에서는 해당 없음 등)는 미집계


5. 구성비 분모 구조 — 핵심 주의사항
──────────────────────────────────────────────────────────────
표14의 구성비는 이중 분모 구조입니다. 이를 이해하지 못하면
KOSIS 수치와 전혀 다른 결과가 나옵니다.

  항목             분모
  ────────────     ───────────────────────────────────────
  자발적 사유       해당 근로형태 전체 (= 계 100 기준)
  비자발적 사유      해당 근로형태 전체 (= 계 100 기준)
  ────────────     ───────────────────────────────────────
  근로조건에 만족   자발적 사유 소계 (= 자발 100 기준)
  안정적인 일자리   자발적 사유 소계
  직장이동 등       자발적 사유 소계
  노력한만큼 수입   자발적 사유 소계
  ────────────     ───────────────────────────────────────
  당장 수입이 필요  비자발적 사유 소계 (= 비자발 100 기준)
  원하는 일자리없음 비자발적 사유 소계
  직장이동 등(비)   비자발적 사유 소계
  노력한만큼(비)    비자발적 사유 소계

  ※ 자주 묻는 질문 ①
     Q. 세부 항목들을 전체 대비로 계산했더니 KOSIS 수치와 크게
        다릅니다.
     A. 표14 세부 항목은 전체 대비가 아닌 자발(또는 비자발) 소계
        대비 비중입니다.
        - 비정규직 자발 비중=67.8%, 근로조건 비중=57.9%
        → 이는 "비정규직 중 근로조건 선택=57.9%"가 아니라
          "비정규직 자발 응답자 중 근로조건 선택=57.9%"입니다.
        - 실제 비정규직 전체 대비 근로조건 비중
          = 67.8% × 57.9% ÷ 100 = 39.3%

  ※ 자주 묻는 질문 ②
     Q. 자발 내 세부 비중 합계가 꼭 100%여야 하나요?
     A. 네, 자발적 사유 소계=100% 안에서 각 세부 항목의 비중 합이
        100이 되어야 합니다. (소수점 반올림으로 99.9~100.1 허용)


6. 집계 절차
──────────────────────────────────────────────────────────────
① 근로형태별 데이터셋 준비 (표준 플래그 생성)

② 각 근로형태 내:
   - 전체 가중합 계산 (총 분모)
   - 자발적 응답자(hsgjno66=1) 가중합 계산 (자발 분모)
   - 비자발적 응답자(hsgjno66=2) 가중합 계산 (비자발 분모)

③ 자발/비자발 비중 = 자발(비자발) 가중합 / 전체 가중합 × 100

④ 세부 항목 비중 (자발 내):
   = 해당 코드 가중합 / 자발 소계 × 100

⑤ 세부 항목 비중 (비자발 내):
   = 해당 코드 가중합 / 비자발 소계 × 100

   예시 (Python — 비정규직 근로조건 비중):
     b2   = wage[b2_flag]
     z    = b2[b2['지난주취업동기코드'] == 1]           # 자발
     zrc  = z[z['지난주취업_주요사유코드'] == 1]        # 근로조건

     wz   = z['가중값'] / 1_000_000
     wzrc = zrc['가중값'] / 1_000_000
     # 자발비중  = round(wz.sum() / w.sum() * 100, 1)  → 67.8%
     # 근로조건  = round(wzrc.sum() / wz.sum() * 100, 1)  → 57.9%

3. 사용 변수 — 근속기간 계산 방법
──────────────────────────────────────────────────────────────
  MDIS 변수명                    SAS 변수명   설명
  ────────────────────────────   ──────────   ────────────────────
  현재일관련사항_직장시작연월     hsgjno31_4   현재 직장 입사연월 (YYYYMM)

  ※ MDIS에 근속기간(개월) 변수가 직접 존재하지 않습니다.
     직장시작연월로부터 역산합니다.

  [근속기간 계산 공식]
  tenure(개월) = (조사년 - 시작년) × 12 + (조사월 - 시작월)

  예시:
    조사연월  = 202508 (2025년 8월)
    직장시작  = 202212 (2022년 12월)
    tenure   = (2025-2022)×12 + (8-12) = 36 - 4 = 32개월

  Python 예시:
    survey_y = 202508 // 100   # → 2025
    survey_m = 202508 %  100   # → 8

    tenure   = (survey_y - start_ym//100)*12 + (survey_m - start_ym%100)


4. 수록 항목별 산출 방법
──────────────────────────────────────────────────────────────
① 평균근속기간 (개월)
  = Σ(tenure × 가중값) / Σ(가중값)
  → round() → 정수 (개월)
  ※ 소수점 없음. 반올림하여 정수 표기

② 증감 (전년동월대비)
  = 당해연도 평균근속기간 - 전년동월 평균근속기간
  → 정수 (개월)

③④⑤ 근속기간별 구성비 (%)
  분모: 해당 근로형태 전체
  분자: 해당 구간에 속하는 인원
  → round(,0.1)

  구간 정의:
    1년 미만:   tenure <  12개월
    1~3년 미만: 12 ≤ tenure < 36개월
    3년 이상:   tenure ≥ 36개월

  ※ 자주 묻는 질문 ①
     Q. 1년을 12개월로 정확히 구분하나요?
     A. 네. 조사연월에서 직장시작연월을 빼서 얻은 월수를
        12 미만/이상으로 구분합니다. 부가조사 기준 월수 기준입니다.


5. 직장시작연월 변수 활용 시 주의사항
──────────────────────────────────────────────────────────────
  ※ 자주 묻는 질문 ②
     Q. MDIS에 근속기간 변수(개월수)가 있지 않나요?
     A. 2025년 기준 MDIS 데이터셋에는 근속기간이 직접 수록된
        변수가 없습니다. 반드시 직장시작연월(hsgjno31_4)로부터
        계산해야 합니다.
        - SAS: hsgjno31_4 (직장시작연월, YYYYMM 형식)
        - Python: '현재일관련사항_직장시작연월'

  ※ 자주 묻는 질문 ③
     Q. tenure=0인 경우를 제외해야 하나요?
     A. 조사 당월에 입사한 경우 tenure=0이 될 수 있습니다.
        KOSIS 집계 기준상 0개월도 포함합니다.
        2025.08 기준 0개월 표본: 약 1,232명 존재
        이를 제외하면 평균근속기간이 소폭 증가하여 불일치 발생

  평균근속기간 가중평균:
